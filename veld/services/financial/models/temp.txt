enum PaymentStatus {
  INITIATED,
  SUCCESS,
  FAILED,
  REFUNDED
}

enum PaymentMethod {
  BANK_TRANSFER,
  CARD,
  DIRECT_DEBIT,
  MOBILE_WALLET
}

enum DisbursementStatus {
  PENDING,
  COMPLETED,
  FAILED
}
enum ApprovalDecision {
  PENDING,
  APPROVED,
  REJECTED
}


enum CollateralType {
  PROPERTY,
  VEHICLE,
  CASH,
  CERTIFICATE_OF_DEPOSIT,
  INVESTMENT,
  GUARANTOR
}

enum CollateralStatus {
  PLEDGED,
  VERIFIED,
  RELEASED,
  LIQUIDATED
}

enum DocumentType {
  ID_PROOF,
  INCOME_PROOF,
  BANK_STATEMENT,
  EMPLOYMENT_LETTER,
  CREDIT_REPORT,
  PROPERTY_DOCUMENT,
  OTHER
}

model LoanPayment {
  id: uuid
  loanId: uuid
  transactionId: uuid
  installmentId: uuid
  amountPaid: decimal
  paymentDate: datetime
  paymentMethod: PaymentMethod
  paymentStatus: PaymentStatus
}

model LoanDisbursement {
  id: uuid
  loanId: uuid
  transactionId: uuid
  amount: decimal
  disbursementDate: datetime
  disbursementStatus: DisbursementStatus
}

model LoanApproval {
  id: uuid
  loanApplicationId: uuid
  approverId: uuid
  decision: ApprovalDecision
  comments: string
  decisionDate: datetime
}

model Collateral {
  id: uuid
  loanId: uuid
  type: CollateralType
  collateralDescription: string
  value: decimal
  valuationDate: date
  status: CollateralStatus
}

model LoanDocument {
  id: uuid
  loanApplicationId: uuid
  documentType: DocumentType
  documentUrl: string
  uploadedAt: datetime
  verified: bool
}
